def set_position_limit(self, account_id: str, limit_amount: float,
                       effective_until: Optional[datetime] = None) -> None:
    # ✅ 统一为PositionService入口

        self.position_manager.set_position_limit(account_id, limit_amount, effective_until)

def get_position_limit(self, account_id: str) -> Optional[PositionLimit]:
    # ✅ 统一为PositionService入口
    if self.position_manager and hasattr(self.position_manager, 'get_position_limit'):
        return self.position_manager.get_position_limit(account_id)
    else:

#### 问题1：L306 _get_tick_size硬编码1.0

**原问题**：`_get_tick_size` 异常时返回硬编码1.0

**当前代码 (L290-319)**：
```python
def _get_tick_size(self, instrument_id: str) -> float:
    try:
        from ali2026v3_trading.params_service import get_params_service
        params_svc = get_params_service()
        meta = params_svc.get_instrument_meta_by_id(instrument_id)
        product = meta.get('product') if meta else instrument_id
        product_cache = params_svc.get_product_cache(product)
        if product_cache:
            tick_size = float(product_cache.get('tick_size', 0))
            if tick_size > 0:
                return tick_size
    except Exception as e:
        logging.debug(f"[OrderService._get_tick_size] Failed to get tick_size from ParamsService: {e}")
    # ✅ P1修复：使用品种配置的默认值，而非硬编码1.0
    default_tick_sizes = {
        'IF': 0.2,  # 沪深300股指期货
        'IC': 0.2,  # 中证500股指期货
        'IH': 0.2,  # 上证50股指期货
        'IM': 0.2,  # 中证1000股指期货
    }
    for prefix, tick in default_tick_sizes.items():
        if instrument_id.startswith(prefix):
            return tick
    return 1.0  # 最终降级
```

**结论**：部分修复。已添加品种配置默认值，但最终降级仍返回1.0。建议扩展 `default_tick_sizes` 字典覆盖更多品种。

#### 问题2：send_order平台下单在锁外

**原问题**：`send_order` 中平台下单在锁外执行，下单期间订单状态可能被其他线程修改

**当前代码 (L141-168)**：
```python
with self._lock:
    self._orders_by_id[order_id] = order
    self._stats['total_orders'] += 1
# 锁外执行平台下单
if self._platform_insert_order and callable(self._platform_insert_order):
    try:
        result = self._platform_insert_order(...)  # 平台下单在锁外
        platform_order_id = self._normalize_platform_result(result).get('order_id')
        if platform_order_id:
            with self._lock:
                order['platform_order_id'] = platform_order_id
                self._platform_id_to_order_id[str(platform_order_id)] = order_id
    except Exception as e:
        with self._lock:
            order['status'] = 'FAILED'
            self._stats['failed_orders'] += 1
```

**结论**：仍存在。平台下单在锁外执行，订单已写入 `_orders_by_id`（状态为'SUBMITTED'），其他线程可读取到该订单。若平台下单失败，状态更新为'FAILED'，但期间其他线程可能基于'SUBMITTED'状态做决策。

**风险等级**：MEDIUM（平台下单通常毫秒级完成，且新创建订单被其他线程访问概率较低）
