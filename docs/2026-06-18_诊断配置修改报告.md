    current_num = cur_year * 12 + cur_month

```python
def _classify_contract_month(year: int, month: int, current_ym: Tuple[int, int]) -> str:
    """月份配置（2026年6月18日设定）：
    - 当月: 2607 (7月)
    - 下月: 2608 (8月)
    - 当季月: 2609 (9月)
    - 下季月: 2612 (12月)
    - 远季月1: 2703 (2026年3月)
    - 远季月2: 2706 (2026年6月)
    """
    ym_str = f"{year % 100:02d}{month:02d}"
    
    if ym_str == '2607':
        return 'current_month'
    if ym_str == '2608':
        return 'next_month'
    if ym_str == '2609':
        return 'current_quarter'
    if ym_str == '2612':
        return 'next_quarter_1'
    if ym_str == '2703':
        return 'next_quarter_2'
    if ym_str == '2706':
        return 'far_quarter'
    
    return 'far'
```

| 分类 | 合约代码 | 月份 | 说明 |
|------|---------|------|------|
| current_month | 2607 | 7月 | 当月合约 |
| next_month | 2608 | 8月 | 下月合约 |
| current_quarter | 2609 | 9月 | 当季月合约 |
| next_quarter_1 | 2612 | 12月 | 下季月合约 |
| next_quarter_2 | 2703 | 2026年3月 | 远季月1 |
| far_quarter | 2706 | 2026年6月 | 远季月2 |
